\begin{algorithm}[t!]
\caption{Algorithm for recovering set $T$ from $\beta$-quantile matrix $\M$.}
\label{alg:recover-T}
\begin{algorithmic}[1]
\State Parameters: number of ratings $k'$, tolerance $\epsilon$
\State Input: matrix $\M$ of approximate $\beta$-quantiles
\State $\robs \gets 0 \in \bR^m$
\For{$k'$ randomly selected indices $j$}
\State $\robs_j \gets $ noisy rating for item $j$
\EndFor
\State Let $\sC$ be the set of $\alpha n$ indices $i \in [n]$ for which 
       $\sum_j \M_{ij}\robs_j$ is largest.
\State $T_0 \gets \frac{1}{|\sC|} \sum_{i \in \sC} \M_i$. \Comment{$T_0 \in [0,1]^m$}
\While{{\bfseries true}}
\State $T \gets \textsc{RandomizedRound}(T_0)$ \Comment{$T \in \{0,1\}^m$}
\If{$\sum_{j} (T_j - (T_0)_j)\robs_j \geq -\epsilon$}
\State Output $T$
\EndIf
\EndWhile
\end{algorithmic}
\end{algorithm}
